      const instruments = {
        piano: { wave: 'triangle', attack: .005, decay: .3, sustain: .4, release: .8, filterFreq: 3000, filterQ: 1 },
        epiano: { wave: 'sine', attack: .01, decay: .5, sustain: .3, release: 1, filterFreq: 2500, filterQ: 2 },
        organ: { wave: 'square', attack: .01, decay: .01, sustain: .8, release: .1, filterFreq: 2000, filterQ: .5 },
        synth: { wave: 'sawtooth', attack: .05, decay: .2, sustain: .5, release: .5, filterFreq: 4000, filterQ: 5 }
      };
      let ctx, analyser, masterGain;
      let currentInst = 'piano', baseOctave = 4, volume = .7, sustainOn = false;
      let activeNotes = {}, recording = [], isRecording = false, recStart = 0, recInterval;

      function initAudio() {
        ctx = new (window.AudioContext || window.webkitAudioContext)();
        analyser = ctx.createAnalyser(); analyser.fftSize = 2048;
        masterGain = ctx.createGain(); masterGain.gain.value = volume;
        masterGain.connect(analyser); analyser.connect(ctx.destination);
      }
      function freq(note, oct) { return 440 * Math.pow(2, (noteNames.indexOf(note) - 9) / 12 + (oct - 4)); }
      function playNote(note, oct) {
        if (!ctx) initAudio();
        const key = note + oct;
        if (activeNotes[key]) return;
        const inst = instruments[currentInst];
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();
        const filter = ctx.createBiquadFilter();
        osc.type = inst.wave; osc.frequency.value = freq(note, oct);
        filter.type = 'lowpass'; filter.frequency.value = inst.filterFreq; filter.Q.value = inst.filterQ;
        const now = ctx.currentTime;
        gain.gain.setValueAtTime(0, now);
        gain.gain.linearRampToValueAtTime(1, now + inst.attack);
        gain.gain.linearRampToValueAtTime(inst.sustain, now + inst.attack + inst.decay);
        osc.connect(filter); filter.connect(gain); gain.connect(masterGain);
        osc.start(now);
        activeNotes[key] = { osc, gain, note, oct };
        if (isRecording) recording.push({ note, oct, time: Date.now() - recStart, type: 'on' });
        const el = document.querySelector(`[data-note="${note}"][data-oct="${oct}"]`);
        if (el) { el.classList.add('pressed'); addGlow(el); addRipple(el); spawnParticles(el); addFallingNote(el, note); }
        updateChord();
      }
      function stopNote(note, oct) {
        const key = note + oct; const n = activeNotes[key]; if (!n) return;
        if (sustainOn) return;

        n.gain.gain.cancelScheduledValues(now);
        n.gain.gain.setValueAtTime(n.gain.gain.value, now);
        n.gain.gain.linearRampToValueAtTime(0, now + inst.release);
        n.osc.stop(now + inst.release + .1);
        delete activeNotes[key];
        if (isRecording) recording.push({ note, oct, time: Date.now() - recStart, type: 'off' });
        const el = document.querySelector(`[data-note="${note}"][data-oct="${oct}"]`);
        if (el) el.classList.remove('pressed');
        updateChord();
      }

        Object.keys(activeNotes).forEach(k => { const n = activeNotes[k]; stopNote(n.note, n.oct); });
      }

      }

              const bw = getComputedStyle(document.documentElement).getPropertyValue('--bk-w') || 30;
              b.style.left = (whiteIdx * 50 + 35) + 'px';
              const bl = document.createElement('span'); bl.className = 'note-label'; bl.textContent = bn;
              const bs = document.createElement('span'); bs.className = 'shortcut-label';
              const bki = whiteIdx; if (keyMapBlack[bki]) bs.textContent = keyMapBlack[bki].toUpperCase();
              b.appendChild(bl); b.appendChild(bs);
              keyboard.appendChild(b);
            }
            whiteIdx++;
          });
        });
        updateScale();
      }

      // Keyboard input mapping
      function getNoteFromKey(k) {
        const wIdx = keyMapWhite.indexOf(k);
        if (wIdx !== -1) {
          const oct = baseOctave + Math.floor(wIdx / 7);
          return { note: whiteNotes[wIdx % 7], oct };
        }
        const bIdx = keyMapBlack.indexOf(k);
        if (bIdx !== -1 && keyMapBlack[bIdx] !== '') {
          const oct = baseOctave + Math.floor(bIdx / 7);
          const wI = bIdx % 7;
          if (wI in blackNoteMap) return { note: blackNoteMap[wI], oct };
        }
        return null;
      }

        const n = getNoteFromKey(k); if (n) playNote(n.note, n.oct);
      });
      document.addEventListener('keyup', e => {
        if (e.key === ' ') { sustainOn = false; releaseSustained(); return; }
        const k = e.key.toLowerCase(); pressedKeys.delete(k);
        const n = getNoteFromKey(k); if (n) stopNote(n.note, n.oct);
      });

      keyboard.addEventListener('pointerleave', e => {

      document.getElementById('volumeSlider').addEventListener('input', e => { volume = e.target.value / 100; if (masterGain) masterGain.gain.value = volume; });
      document.getElementById('bpmSlider').addEventListener('input', e => { metroBPM = parseInt(e.target.value); document.getElementById('bpmDisplay').textContent = metroBPM; if (metroOn) startMetro(); });

        if (!isRecording) {

          recInterval = setInterval(() => { const s = Math.floor((Date.now() - recStart) / 1000); recTimer.textContent = String(Math.floor(s / 60)).padStart(2, '0') + ':' + String(s % 60).padStart(2, '0'); }, 200);
        } else { isRecording = false; recBtn.classList.remove('active'); clearInterval(recInterval); recTimer.style.display = 'none'; }
      });

      const playBtn = document.getElementById('playBtn');
      const progBar = document.getElementById('progressBar');

        if (!sc || !scales[sc]) return;
        const intervals = scales[sc];

      // Chord detection
      const chordPatterns = {
        '': [0, 4, 7], 'm': [0, 3, 7], 'dim': [0, 3, 6], 'aug': [0, 4, 8],
        '7': [0, 4, 7, 10], 'maj7': [0, 4, 7, 11], 'm7': [0, 3, 7, 10],
        'sus2': [0, 2, 7], 'sus4': [0, 5, 7]
      };
      function updateChord() {
        const notes = Object.values(activeNotes).map(n => noteNames.indexOf(n.note));
        if (notes.length < 2) { chordDisplay.textContent = ''; return; }
        const unique = [...new Set(notes.map(n => n % 12))].sort((a, b) => a - b);
        for (let root of unique) {
          const intervals = unique.map(n => (n - root + 12) % 12).sort((a, b) => a - b);
          for (let [name, pat] of Object.entries(chordPatterns)) {
            if (pat.length === intervals.length && pat.every((v, i) => v === intervals[i])) {
              chordDisplay.textContent = noteNames[root] + name; return;
            }
          }
        }
        chordDisplay.textContent = Object.values(activeNotes).map(n => n.note).join(' + ');
      }

      function spawnParticles(el) {
        const rect = el.getBoundingClientRect();
        for (let i = 0; i < 6; i++) {
          particles.push({
            x: rect.left + rect.width / 2, y: rect.top,
            vx: (Math.random() - .5) * 4, vy: -Math.random() * 5 - 2,
            life: 1, symbol: noteSymbols[Math.floor(Math.random() * noteSymbols.length)],
            color: `hsl(${Math.random() * 360},80%,65%)`, size: 10 + Math.random() * 10
          });
        }
      }

      function addFallingNote(el, note) {

        fallingNotes.push({ x: rect.left + rect.width / 2 - 8, y: 0, w: 16, h: 20 + Math.random() * 30, color: noteColors[ni % noteColors.length], vy: 2 + Math.random() * 2, life: 1 });

      function resizeCanvases() {
        [fallingCanvas, particleCanvas].forEach(c => { c.width = window.innerWidth; c.height = window.innerHeight; });
        waveCanvas.width = waveCanvas.offsetWidth; waveCanvas.height = waveCanvas.offsetHeight;

      // Animation loop
      function animate() {
        requestAnimationFrame(animate);
        // Falling notes
        fCtx.clearRect(0, 0, fallingCanvas.width, fallingCanvas.height);
        fallingNotes.forEach((n, i) => {
          n.y += n.vy; n.life -= .003;
          fCtx.globalAlpha = Math.max(0, n.life);
          fCtx.fillStyle = n.color;
          fCtx.shadowColor = n.color; fCtx.shadowBlur = 10;
          fCtx.beginPath(); fCtx.roundRect(n.x, n.y, n.w, n.h, 4); fCtx.fill();
          fCtx.shadowBlur = 0;
        });
        fallingNotes = fallingNotes.filter(n => n.life > 0 && n.y < fallingCanvas.height);
        // Particles
        pCtx.clearRect(0, 0, particleCanvas.width, particleCanvas.height);
        particles.forEach(p => {
          p.x += p.vx; p.y += p.vy; p.vy += .1; p.life -= .02;
          pCtx.globalAlpha = Math.max(0, p.life);
          pCtx.fillStyle = p.color; pCtx.font = p.size + 'px serif';
          pCtx.fillText(p.symbol, p.x, p.y);
        });
        particles = particles.filter(p => p.life > 0);
        // Waveform
        if (analyser) {
          const buf = new Uint8Array(analyser.frequencyBinCount);
          analyser.getByteTimeDomainData(buf);
          wCtx.clearRect(0, 0, waveCanvas.width, waveCanvas.height);
          wCtx.lineWidth = 2; wCtx.strokeStyle = getComputedStyle(document.body).getPropertyValue('--accent') || '#6c63ff';
          wCtx.beginPath();
          const sl = waveCanvas.width / buf.length; let x = 0;
          for (let i = 0; i < buf.length; i++) {
            const v = buf[i] / 128; const y = v * waveCanvas.height / 2;
            i === 0 ? wCtx.moveTo(x, y) : wCtx.lineTo(x, y); x += sl;
          }
          wCtx.lineTo(waveCanvas.width, waveCanvas.height / 2); wCtx.stroke();
        }
      }

      function positionBlackKeys() {
        const whites = keyboard.querySelectorAll('.key.white');
        const blacks = keyboard.querySelectorAll('.key.black');
        blacks.forEach(b => {
          const bn = b.dataset.note, bo = parseInt(b.dataset.oct);
          let prevWhite = null;
          whites.forEach(w => {
            const wn = w.dataset.note, wo = parseInt(w.dataset.oct);
            if (wo === bo) {
              if ((bn === 'C#' && wn === 'C') || (bn === 'D#' && wn === 'D') || (bn === 'F#' && wn === 'F') || (bn === 'G#' && wn === 'G') || (bn === 'A#' && wn === 'A'))
                prevWhite = w;
            }
          });
          if (prevWhite) {
            const pRect = prevWhite.offsetLeft; const pw = prevWhite.offsetWidth;
            b.style.left = (pRect + pw - 15) + 'px';
          }
        });
      }
